    MouseGetPos tx, ty

BankCapBuyer:
    {
        Critical on
        BankCapBuyerIcon.toggle()
        if !(BankCapBuyerIcon.isactive) {
            SetTimer, bankpicker, off
        }
        else {
            SetTimer, bankpicker, %checkBanksFullDelay%
            bankpicker()
        } 
        return
    }
