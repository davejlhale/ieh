

lootBestiary:
    {
                Critical on

        lootBestiaryIcon.toggle()
        if !(lootBestiaryIcon.isactive) {
            SetTimer, timedlootBestiary, off
            return
        }
        else {
            ;call this label on interval - 120000000 = 20mins
            SetTimer, timedlootBestiary, 120000000 
        } 
        timedlootBestiary:
            BlockInput mousemove
            MouseGetPos tx, ty
            
            gClick(menu3,2,100)
            gclick(explore_bestiary,2,100)
            gclick(bestiary_lootAll,2,100)
            gclick(bestiary_close,2,200)
            
            mousemove tx,ty
            BlockInput mousemoveoff
        return
    }
